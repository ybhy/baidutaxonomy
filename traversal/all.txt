                if (div.GetAttribute("className").Equals("main-content"))
                {
                    foreach (HtmlElement a in div.GetElementsByTagName("A"))
                    {
                        if (a.GetAttribute("target").Equals("_blank"))
                        {
                            if (a.InnerText != null)
                            {
                                entity = a.InnerText;
